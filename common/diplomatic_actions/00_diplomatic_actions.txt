# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

# Note:
#   The action is always available, so potential is which instance we want to limit, and allow is whether it is allowed or not
#   Therefore, don't add checks that assume failure means the action is blocked, the checks must pass to block the action

# Diplomatic Actions
allianceaction = {
    condition = {
        tooltip = WOW_TROLL_CONDEMNED_THE_OLD_GODS
        potential = {
            technology_group = tech_troll
            FROM = { religion = religion_old_gods }
        }
        allow = {
            NOT = { has_country_modifier = condemned_the_old_gods }
        }
    }
    condition = {
        tooltip = WOW_OLD_GOD_CONDEMNED_THE_OLD_GODS
        potential = {
            FROM = { technology_group = tech_troll }
            religion = religion_old_gods
        }
        allow = {
            FROM = { NOT = { has_country_modifier = condemned_the_old_gods } }
        }
    }
    condition = {
        tooltip = WOW_ALIGNMENT_INCOMPATIBLE
        potential = {
            is_good = yes
        }
        allow = {
            FROM = { is_good = yes }
        }
    }
    condition = {
        tooltip = WOW_ALIGNMENT_INCOMPATIBLE
        potential = {
            is_evil = yes
        }
        allow = {
            FROM = { is_evil = yes }
        }
    }
}

declarewar = {
    condition = {
        tooltip = WOW_WAR_ON_ALLIANCE
        potential = {
            has_country_modifier = alliance_member
            FROM = { has_country_modifier = alliance_member }
        }
        allow = {
            always = no
        }
    }
    condition = {
        tooltip = WOW_WAR_ON_HORDE
        potential = {
            has_country_modifier = horde_member
            FROM = { has_country_modifier = horde_member }
        }
        allow = {
            always = no
        }
    }
    condition = {
        tooltip = WOW_WAR_ON_NEUTRAL
        potential = {
            has_country_modifier = neutral_member
        }
        allow = {
            NOT = {
              has_country_modifier = alliance_member
              has_country_modifier = horde_member
              has_country_modifier = neutral_member
            }
        }
    }
    condition = {
        tooltip = WOW_WAR_FROM_NEUTRAL
        potential = {
            FROM = { has_country_modifier = neutral_member }
        }
        allow = {
            NOT = {
              has_country_modifier = alliance_member
              has_country_modifier = neutral_member
            }
        }
    }
}
