# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

#   Therefore, don't add checks that assume failure means the action is blocked, the checks must pass to block the action

# Diplomatic Actions
allianceaction = {
    condition = {
        tooltip = WOW_DIPLOMATICALLY_INCOMPATIBLE
        
        potential = {
            OR = {
                AND = {
                    is_good = yes
                    FROM = { 
                        NOT = { is_good = yes }
                    }
                }
                AND = {
                    is_evil = yes
                    FROM = { 
                        NOT = { is_evil = yes }
                    }
                }
                AND = {
                    is_civilized = yes
                    FROM = { 
                        NOT = { is_civilized = yes }
                    }
                }
                AND = {
                    is_savage = yes
                    FROM = { 
                        NOT = { is_savage = yes }
                    }
                }
                AND = {
                    is_lesser = yes
                    FROM = { 
                        OR = {
                            AND = {
                                is_lesser = yes
                                NOT = { is_neighbor_of = FROM }
                            }
                            NOT = { is_lesser = yes }
                        }
                    }
                }
                AND = {
                    is_bug = yes
                    FROM = { 
                        NOT = { is_bug = yes }
                    }
                }
                AND = {
                    is_pure_evil = yes
                    FROM = { 
                        NOT = { 
                            OR = {
                                is_pure_evil = yes
                                is_savage = yes
                                is_lesser = yes
                                is_bug = yes
                            }
                        }
                    }
                }
                AND = {
                    technology_group = tech_dragonkin
                    FROM = { 
                        NOT = { technology_group = tech_dragonkin }
                    }
                }
                AND = {
                    technology_group = tech_elemental
                    FROM = { 
                        NOT = { technology_group = tech_elemental }
                    }
                }
            }
        } 
        allow = {
            always = no
        }
    }
}
