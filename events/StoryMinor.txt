# Balnazzar's Reveal
country_event = {
	id = story.1
	title = story.1.title
	desc = story.1.desc
	picture = NATHREZIM_eventPicture

	major = yes

	trigger = {
		tag = A22 # Scarlet Crusade
		OR = {
			NOT = { stability = -2 }
			dominant_culture = culture_forsaken
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = story.1.option.a

		change_tag = X01 # The Risen
		change_government = scourge_gov
		change_primary_culture = culture_forsaken
		change_religion = religion_fel
		every_owned_province = {
			change_culture = culture_forsaken
			change_religion = religion_fel
		}
		add_stability = 6

		define_ruler = {
			name = "Balnazzar"
			dynasty = "Dreadlord"
			ADM = 3
			DIP = 1
			MIL = 6
			attach_leader = "Balnazzar"
		}
	}
}

# Insidious Happenings
country_event = {
  id = story.2
  title = story.2.title
  desc = story.2.desc
  picture = FORSAKEN_eventPicture

  trigger = {
    tag = A22 # Scarlet Crusade
    NOT = { stability = 0 }
    any_owned_province = {
      is_capital = no
      NOT = { culture = culture_forsaken }
    }
  }

  mean_time_to_happen = {
    months = 120
  }

  option = {
    name = story.2.option.a

    random_owned_province = {
      limit = { is_capital = no }

      change_culture = culture_forsaken
    }
  }
}
