# Random death due to disease
# [Root.Monarch.GetTitle] [Root.Monarch.GetName] falls ills
country_event = {
  id = wow_ruler.1
  title = wow_ruler.1.title
  desc = wow_ruler.1.desc
  picture = BURNING_CASTLE_eventPicture

  trigger = {
    NOT = { culture_group = culture_group_demonic }
    NOT = { culture_group = culture_group_dragonkin }
    NOT = { culture_group = culture_scourge }
    NOT = { culture_group = culture_crypt_fiends }
    NOT = { culture_group = culture_ghost }
    NOT = { culture_group = culture_undead }
    NOT = { culture_group = culture_group_titan }
    NOT = { culture_group = culture_group_northrend_titan }
    NOT = { culture_group = culture_group_fire }
    NOT = { culture_group = culture_group_earth }
    NOT = { culture_group = culture_group_water }
    NOT = { culture_group = culture_group_air }
    NOT = { culture_group = culture_group_nature }
    NOT = { culture_group = culture_group_sha }
    NOT = { culture_group = culture_group_mogu }
    NOT = { culture_group = culture_group_ethereal }
    NOT = { culture_group = culture_group_mecha_gnomish }
    NOT = { culture_group = culture_group_mecha_ice_giant }
    NOT = { culture_group = culture_group_earth_giant }
    NOT = { culture_group = culture_group_dryad }
    NOT = { culture = culture_qiraji }
    has_regency = no
  }

  option = {
    name = wow_ruler.1.option.a

    kill_ruler = yes
  }
  option = {
    name = wow_ruler.1.option.b

    add_years_of_income = -1

    random = {
      chance = 50
      kill_ruler = yes
    }
  }
}

# For races that have lifespans that exceed 100 years
# [Root.Monarch.GetTitle] [Root.Monarch.GetName] passed away
country_event = {
  id = wow_ruler.3
  title = wow_ruler.3.title
  desc = wow_ruler.3.desc
  picture = BURNING_CASTLE_eventPicture

  trigger = {
    has_regency = no
    OR = {
      AND = {
        NOT = { culture_group = culture_group_demonic }
        NOT = { culture_group = culture_group_draenei }
        NOT = { culture_group = culture_group_dragonkin }
        NOT = { culture_group = culture_group_earth_giant }
        NOT = { culture_group = culture_group_dryad }
        NOT = { culture_group = culture_group_scourge }
        NOT = { culture_group = culture_group_crypt_fiends }
        NOT = { culture_group = culture_group_ghost }
        NOT = { culture_group = culture_group_undead }
        NOT = { culture_group = culture_group_titan }
        NOT = { culture_group = culture_group_northrend_titan }
        NOT = { culture_group = culture_group_fire }
        NOT = { culture_group = culture_group_earth }
        NOT = { culture_group = culture_group_water }
        NOT = { culture_group = culture_group_air }
        NOT = { culture_group = culture_group_nature }
        NOT = { culture_group = culture_group_sha }
        NOT = { culture_group = culture_group_mogu }
        NOT = { culture_group = culture_group_ethereal }
        NOT = { culture_group = culture_group_mecha_gnomish }
        NOT = { culture_group = culture_group_ice_giant }
        NOT = { culture = culture_qiraji }
        OR = {
          AND = {
            OR = {
              culture_group = culture_group_dwarven
              culture_group = culture_group_pandaren
              culture_group = culture_group_gnomish
            }
            ruler_age = 250
          }
          AND = {
            culture_group = culture_group_thalassian
            ruler_age = 350
          }
          AND = {
            OR = {
              culture_group = culture_group_darnassian
              culture_group = culture_group_sea_giant
              culture_group = culture_group_satyr
              culture_group = culture_group_naga
            }
            ruler_age = 1200
          }
        }
      }
    }
  }

  option = {
    name = wow_ruler.3.option.a

    kill_ruler = yes
  }
}

# ADM skill decays...
country_event = {
  id = wow_ruler.4
  title = wow_ruler.4.title
  desc = wow_ruler.4.desc
  picture = BURNING_CASTLE_eventPicture

  trigger = {
    has_regency = no
    ADM = 1
  }

  mean_time_to_happen = {
    months = 1200

    # Rarely fire for short lived rulers
    modifier = {
      factor = 10
      NOT = { ruler_age = 60 }
    }

    modifier = {
      factor = 2
      NOT = { ADM = 3 }
    }

    modifier = {
      factor = 2
      NOT = { ADM = 2 }
    }

    modifier = {
      factor = 0.5
      ADM = 4
    }

    modifier = {
      factor = 0.5
      ADM = 5
    }

    modifier = {
      factor = 0.5
      ADM = 6
    }
  }

  option = {
    name = wow_ruler.4.option.a

    change_adm = -1
  }
}
